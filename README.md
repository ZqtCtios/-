# 这是一个在命令行下的课表小软件
数据和配置用json文件存储，方便修改
json文件说明
```json
{
    "week_data": {//星期数据
        "long": "45",//课程时长min
        "time": {//每节课开始时间
            "1": "8:00",
            "2": "9:00",
            "3": "10:00",
            "4": "11:00",
            "5": "13:00",
            "6": "14:00",
            "7": "15:00",
            "8": "16:00",
            "9": "18:00",
            "10": "19:00"
        }
    },
    "course_data": {//课程数据
        "num": 1,//总课程数
        "courses": {
            "1": {//课程
                "cname": "数据结构",
                "cfrom": "1",//从第几节课开始
                "cto=0": "2",//到第几节课结束
                "weekkind": "1",//上课模式1-单周。2-双周，3-全周
                "week_num": "",//星期几上课
                "week_from": "",//从第几周开始
                "week_to": "",//从第几周开始上课
                "teacher": "",//到第几周结束
                "classroom": ""//教室
            },
            
        }
    },
    "homeWk_dir": "homework"//作业地址
}
```
